#include using namespace std; const int MAX = 1e6 + 5; vector phi(MAX); // To store the totient function for each number // Function to calculate Euler's Totient function for all numbers from 1 to MAX void computeTotient() { for (int i = 1; i < MAX; i++) { phi[i] = i; // Initialize phi[i] with i } for (int i = 2; i < MAX; i++) { if (phi[i] == i) { // i is a prime number for (int j = i; j < MAX; j += i) { phi[j] = (phi[j] * (i - 1)) / i; // Applying the formula } } } } int main() { ios::sync_with_stdio(false); cin.tie(0); // Precompute Euler's Totient function for all numbers up to MAX computeTotient(); int t; cin >> t; while (t--) { int a, b, q; cin >> a >> b >> q; vector queries(q); for (int i = 0; i < q; i++) { cin >> queries[i]; } for (int x : queries) { if (a % x != 0) { cout << 0 << " "; // If x does not divide a, the answer is 0 } else { int a_prime = a / x; int b_prime = b / x; cout << phi[a_prime] << " "; // Print the result for this query } } cout << "\n"; // New line after each test case } return 0; }
